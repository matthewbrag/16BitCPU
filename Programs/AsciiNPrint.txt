//print string from GP memory n times

//input number of times
INP 0 r0

//r3 keeps track of times looped
LD 0 r3

LABEL main_loop

//exit program when we are done
INOP SUB rf r3 r0
JCN ZERO end

//inc r3

INOP ADD r3 r3 1

//string pointer
LD 0 r1

LABEL print_char
GPLD r1 r2

//if the character is null, restart loop
INOP SUB rf r2 0
JCN ZERO main_loop

OUT 0 r2

JMP print_char

LABEL end
HLT
